```dataviewjs
// --- 📅 Papers per Month (Mermaid Line Chart) ---
let pages = dv.pages('"Research_Notes"').sort(p => p.publication_date);
let dateCounts = {};

for (let p of pages) {
    if (p.publication_date) {
        // Format: YYYY-MM
        let dateStr = p.publication_date.toString(); 
        // Handle different date formats if necessary, assuming YYYY-MM-DD
        let month = dateStr.substring(0, 7); 
        dateCounts[month] = (dateCounts[month] || 0) + 1;
    }
}

let sortedMonths = Object.keys(dateCounts).sort();
let trendData = sortedMonths.map(m => dateCounts[m]).join(", ");
let trendLabels = sortedMonths.map(m => `"${m}"`).join(", ");

let mermaidCode = `
%%{init: {'theme': 'base', 'themeVariables': { 'primaryColor': '#ff9f43', 'edgeLabelBackground':'#fff'}}}%%
xychart-beta
    title "Monthly Paper Intake"
    x-axis [${trendLabels}]
    y-axis "Papers" 0 --> ${Math.max(...Object.values(dateCounts)) + 2}
    line [${trendData}]
`;

dv.paragraph('```mermaid\n' + mermaidCode + '\n```');
```

## 🍩 Topic Distribution

```dataviewjs
// --- 🏷️ Tag Distribution (Mermaid Pie Chart) ---
let pages = dv.pages('"Research_Notes"');
let tagCounts = {};

for (let p of pages) {
    if (p.tags) {
        let tags = Array.isArray(p.tags) ? p.tags : [p.tags];
        for (let t of tags) {
            if (typeof t === 'string' && t !== "paper" && !t.startsWith("Score_")) {
                tagCounts[t] = (tagCounts[t] || 0) + 1;
            }
        }
    }
}

// Top 5 Tags + Others
let sortedTags = Object.entries(tagCounts).sort((a, b) => b[1] - a[1]);
let topTags = sortedTags.slice(0, 5);
let otherCount = sortedTags.slice(5).reduce((acc, curr) => acc + curr[1], 0);

let pieData = topTags.map(t => `"${t[0]}" : ${t[1]}`).join("\n    ");
if (otherCount > 0) pieData += `\n    "Others" : ${otherCount}`;

let mermaidPie = `
pie title Top Research Topics
    ${pieData}
`;

dv.paragraph('```mermaid\n' + mermaidPie + '\n```');
```

```dataviewjs
// --- 🏅 High Score Table with Progress Bars ---
let pages = dv.pages('"Research_Notes"')
    .where(p => p.score >= 7)
    .sort(p => p.score, 'desc')
    .limit(10);

dv.table(
    ["Paper Title", "Score", "Innovation Snippet"],
    pages.map(p => {
        // Create HTML Progress Bar
        let color = p.score >= 9 ? "#ff4757" : "#ffa502";
        let bar = `<progress value="${p.score}" max="10" style="width: 80px; accent-color: ${color};"></progress> <b>${p.score}</b>`;
        
        // Link
        let link = `[[${p.file.path}|${p.aliases ? p.aliases[0] : p.file.name}]]`;
        
        return [link, bar, p.innovation ? p.innovation.substring(0, 80) + "..." : ""];
    })
);
```
